    names = {'f3_200.mat', 'f3_250.mat'};
    for i = 1:2
        % init variables to be loaded into simulink model
        T = 0.0001;
        Ka = 1;
        Kt = 0.49;
        Ke = 1.59;

        Bx = 0.0094;
        Jy = 0.0003;

        T_ = 0.0021323;
        Kx = 0.75858;
        Ky = 0.82224;
