low feedrate

name = 'lli.mat';
lli = load(name);
time = lli.lli.X.Data;
x_act = lli.lli.Y(1).Data;
y_act = lli.lli.Y(2).Data;
x_ref = lli.lli.Y(3).Data;
y_ref = lli.lli.Y(4).Data;

e_x_lowFR = x_ref - x_act;
e_y_lowFR = y_ref - y_act;

clf();
hold on;
plot(time, e_x_lowFR);
plot(time, e_y_lowFR);
